#### 6.手动配置文件
```
{
  "log": {
    "loglevel": "warning"
  },

  "inbound": {
    "port": 1081,
    "listen": "127.0.0.1",
    "protocol": "socks",
    "settings": {
      "auth": "noauth",
      "udp": false,
      "ip": "127.0.0.1"
    }
  },

  "outbound": {
    "protocol": "vmess",

    "settings": {
      "vnext": [{
        "address": "svip21.mm1080p.rocks",
        "port": 10086,
        "users": [{
          "id": " ",
          "alterId": 10
        }]
      }]
    },

    "streamSettings": {
      "network": "tcp",
      "security": "tls"
    }
  },

  "policy": {
    "levels": {
      "0": {
        "uplinkOnly": 0
      }
    }
  }
}
```
